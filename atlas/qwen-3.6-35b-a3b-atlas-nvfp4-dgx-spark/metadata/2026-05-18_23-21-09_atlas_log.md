# Docker compose 
```yml
# Atlas inference setup for Qwen3.6-35B-A3B NVFP4 on DGX Spark (GB10)
#
# Reference post by Azeez (Atlas team):
#   https://x.com/AtlasInference/status/2055740933057007745?s=20
# Atlas GitHub: https://github.com/Avarok-Cybersecurity/atlas

services:
  qwen-atlas-nvfp4:
    image: avarok/atlas-gb10:latest
    container_name: qwen-3-6-35b-a3b-nvfp4-atlas
    hostname: inference-server
    platform: linux/arm64

    ports:
      - "8000:8000"

    volumes:
      - ~/.cache/huggingface:/root/.cache/huggingface
      - /dev/shm:/dev/shm

    shm_size: "32g"
    ipc: host

    deploy:
      resources:
        reservations:
          devices:
            - capabilities: [gpu]

    environment:
      HF_HUB_ENABLE_HF_TRANSFER: "1"

    command:
      # Serve the Qwen3.6-35B-A3B NVFP4 model via Atlas inference engine
      # Vanilla config from: https://x.com/AtlasInference/status/2055740933057007745?s=20
      - "serve"
      - "${ATLAS_MODEL_ID:-RedHatAI/Qwen3.6-35B-A3B-NVFP4}"
      - "--model-name"
      - "${ATLAS_MODEL_NAME:-qwen3.6-35b}"

      # --- API & NETWORKING ---
      # --bind 0.0.0.0: Atlas listens on ALL interfaces inside the container.
      #   This is REQUIRED because the model runs in a Docker container, so clients
      #   do NOT reach it via 127.0.0.1 like they would on a local gaming GPU.
      #   The nginx reverse proxy (nginx-proxy) forwards requests here over the
      #   shared Docker network using the hostname "inference-server" set above.
      # SECURITY NOTE: --bind only controls which NIC Atlas binds to inside the
      #   container, NOT which hosts can reach it. To restrict external access,
      #   remove the port mapping below so port 8000 is only reachable from other
      #   containers on the Docker network (i.e., nginx-proxy). With port mapping
      #   present, any host that can reach this machine's IP on port 8000 will
      #   connect directly — rely on --require-auth / --auth-token as gate.
      - "--bind"
      - "0.0.0.0"
      - "--port"
      - "8000"

      # Atlas uses --auth-token (single bearer token). Token leaks via ps but
      # that's fine for dev — use --auth-tokens-file in production.
      # Value comes from ATLAS_API_KEY in .env (gitignored, see .env.example).
      - "--require-auth"
      - "--auth-token"
      - "${ATLAS_API_KEY:-dummy-key}"

      # --- KV CACHE & QUANTIZATION ---
      # --kv-cache-dtype nvfp4: Uses NVFP4 (E2M1 4-bit) for KV cache. This is the
      #   best-performing setting on Blackwell GB10, as shown in the Atlas QUICKSTART.md
      #   top model config (Sehyo/Qwen3.5-35B-A3B-NVFP4 at 133 tok/s). Our original
      #   fp8 setting triggered warnings: "BF16 values are silently clipped to E4M3
      #   range [-448, 448], destroying dynamic range". NVFP4 avoids this entirely.
      #   Reference: atlas/crates/spark-server/src/cli.rs:60-63 — "NVFP4 uses less
      #   memory but may lose coherence at long context without --kv-high-precision-layers"
      - "--kv-cache-dtype"
      - "nvfp4"

      # --kv-high-precision-layers auto: Keeps first 2 + last 2 attention layers in
      #   BF16 KV cache precision, protecting attention sink tokens and output quality
      #   from quantization error while saving memory on middle layers. Without this
      #   (default=0), Atlas warns: "NVFP4 models may hallucinate or lose coherence
      #   at long context." The small overhead of 4 BF16 layers is worth the quality gain.
      #   Reference: atlas/crates/spark-server/src/cli.rs:65-71 —
      #   "Accepts: number, 'auto' (=2, recommended), 'max'/'all' (all BF16)."
      - "--kv-high-precision-layers"
      - "auto"

      # --gpu-memory-utilization: Fraction of GPU VRAM for model + KV cache.
      #   0.65 = 65% of GB10 memory (~78GB), leaving headroom for system overhead.
      #   Tuned down from 0.88 (Atlas QUICKSTART default) for DGX Spark stability.
      - "--gpu-memory-utilization"
      - "0.65"

      # --max-seq-len: Maximum sequence length in tokens.
      #   131072 = ~128K tokens for long context windows.
      - "--max-seq-len"
      - "131072"

      # --- SCHEDULING & TOOLS ---
      # --mtp-quantization nvfp4: Quantizes MTP draft head weights to NVFP4 for fused
      #   device-side expert dispatch. This is the recommended/fastest setting when
      #   --speculative is enabled with NVFP4 models. Alternatives: fp8 (slower, D2H
      #   sync in MoE), bf16 (highest accuracy, most memory).
      #   Reference: atlas/crates/spark-server/src/cli.rs:150-153 — "nvfp4 (fastest,
      #   recommended — uses fused device-side expert dispatch)"
      - "--mtp-quantization"
      - "nvfp4"

      # --scheduling-policy slai: Shared-latency-aware inference scheduling.
      # --tool-call-parser qwen3_coder: Required for tool calling with Qwen3 models.
      # --enable-prefix-caching: Reuse KV cache across requests with common prefixes.
      # --speculative: Enable MTP speculative decoding for faster token generation.
      - "--scheduling-policy"
      - "slai"
      - "--tool-call-parser"
      - "qwen3_coder"
      - "--enable-prefix-caching"
      - "--speculative"

    networks:
      - development-network

networks:
  development-network:
    external: true
```
